#include<bits/stdc++.h>
 
using namespace std;
#define ll long long int

 
 
 
int main()
{     ios_base::sync_with_stdio(false);
      cin.tie(NULL);
    ll n;
    cin >> n;
    ll q;
      cin >> q;
    ll a[n];
    for(int i=0;i<n;i++)
      {
          cin >> a[i];
          i>0?a[i]+=a[i-1]:a[i];
        }
      
      while(q--)
      {
        ll x;
        cin >> x;
        ll index=lower_bound(a,a+n,x)-a;
          cout <<index +1<< endl;
      }
      
 
 
}
